```sql
-- 1. Repositories Table
CREATE TABLE repositories (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    project_key VARCHAR(255),
    slug_name VARCHAR(255),
    clone_url TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(project_key, slug_name)
);

-- 2. Commits Table
CREATE TABLE commits (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    repository_id INTEGER,
    commit_hash VARCHAR(40) UNIQUE,
    author_name VARCHAR(255),
    author_email VARCHAR(255),
    commit_date DATETIME,
    commit_message TEXT,
    lines_added INTEGER DEFAULT 0,
    lines_deleted INTEGER DEFAULT 0,
    files_changed INTEGER DEFAULT 0,
    branch_name VARCHAR(255),
    FOREIGN KEY (repository_id) REFERENCES repositories(id)
);

-- 3. Pull Requests Table
CREATE TABLE pull_requests (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    repository_id INTEGER,
    pr_number INTEGER,
    title TEXT,
    description TEXT,
    author_name VARCHAR(255),
    author_email VARCHAR(255),
    created_date DATETIME,
    updated_date DATETIME,
    merged_date DATETIME,
    state VARCHAR(50),
    source_branch VARCHAR(255),
    destination_branch VARCHAR(255),
    FOREIGN KEY (repository_id) REFERENCES repositories(id),
    UNIQUE(repository_id, pr_number)
);

-- 4. PR Approvals Table
CREATE TABLE pr_approvals (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    pull_request_id INTEGER,
    approver_name VARCHAR(255),
    approver_email VARCHAR(255),
    approval_date DATETIME,
    status VARCHAR(50),
    FOREIGN KEY (pull_request_id) REFERENCES pull_requests(id)
);

-- 5. Staff Details Table (71 fields!)

    id INTEGER PRIMARY KEY AUTOINCREMENT,
    bank_id_1 VARCHAR(50) UNIQUE,
    staff_id VARCHAR(50),
    staff_name VARCHAR(255),
    email_address VARCHAR(255),
    rank VARCHAR(100),
    reporting_manager_name VARCHAR(255),
    work_location VARCHAR(255),
    staff_type VARCHAR(100),
    staff_status VARCHAR(50),
    tech_unit VARCHAR(255),
    -- ... 61 more fields
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 6. Author-Staff Mapping Table
CREATE TABLE author_staff_mapping (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    author_name VARCHAR(255) UNIQUE,
    author_email VARCHAR(255),
    bank_id_1 VARCHAR(50),
    staff_id VARCHAR(50),
    staff_name VARCHAR(255),
    mapped_date DATETIME DEFAULT CURRENT_TIMESTAMP,
    notes TEXT,
    FOREIGN KEY (bank_id_1) REFERENCES staff_details(bank_id_1)
);

-- Indexes for Performance
```
